128  void resize(const int newSize) {
129  if (newSize <= maxSize) {
130  // enough elements is already allocated
131  if (newSize >= size) {
132  // enlarging array, we need to construct some new elements
133  for (int i = size; i < newSize; ++i) {
134  new (data + i) T();
135  }
136  } else {
137  // shrinking array, we need to delete some elements
138  for (int i = newSize; i < size; ++i) {
139  data[i].~T();
140  }
141  }
142  } else {
143  // requested more elements than allocate, need to reallocated.
144  // allocate twice the current number or the new size, whatever is higher, to avoid frequent
145  // reallocation when pushing elements one by one
146  const int actNewSize = Math::max(2 * maxSize, newSize);
147  Array newArray(0, actNewSize);
148  // copy all elements into the new array, using move constructor
149  for (int i = 0; i < size; ++i) {
150  new (newArray.data + i) T(std::move(this->data[i]));
151  }
152  // default-construct new elements
153  for (int i = size; i < newSize; ++i) {
154  new (newArray.data + i) T();
155  }
156  // move the array into this
157  *this = std::move(newArray);
158  }
159  this->size = newSize;
160  }
