33  auto integrate(TFunctor&& f, const Float targetError = 0.001_f) {
34  double sum = 0.;
35  double sumSqr = 0.;
36  double errorSqr = Math::sqr(targetError);
37  int64_t n = 0;
38  StaticField<chunk> buffer;
39  StaticArray<bool, chunk> inside;
40  Vector center = domain->getCenter();
41  Float radius = domain->getBoundingRadius();
42  while (true) {
43  for (int i = 0; i < chunk; ++i) {
44  // dimensionless vector
46  const Vector q(rng(0), rng(1), rng(2));
47  // vector with units and properly scaled and moved
48  const Vector v = radius * (2._f * q - Vector(1._f)) + center;
49  buffer[i] = v;
50  }
51  domain->areInside(buffer, inside);
52  for (int i = 0; i < chunk; ++i) {
53  if (inside[i]) {
54  const double x = (double)f(buffer[i]);
55  sum += x;
56  sumSqr += x * x;
57  n++;
58  }
59  }
60  // const TResult e = (n*sumSqr - sum*sum)/(n*sumSqr);
61  // std::cout << e << std::endl;
62  // standard deviation = error = sqrt(variance) !!!
63  // std::cout << "ERROR = " << 1.f/n* (n*sumSqr - sum*sum) / (n*sumSqr) << std::endl;
64  const double m = double(n);
65  if (m * sumSqr - sum * sum < m * m * errorSqr * sumSqr) {
66  return Float(sum / m) * domain->getVolume();
67  }
68  }
