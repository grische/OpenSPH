78  template <typename DistanceType, typename IndexType = size_t, typename CountType = size_t>
79  class KNNResultSet {
80  IndexType* indices;
81  DistanceType* dists;
82  CountType capacity;
83  CountType count;
84 
85  public:
86  inline KNNResultSet(CountType capacity_)
87  : indices(0)
88  , dists(0)
89  , capacity(capacity_)
90  , count(0) {}
91 
92  inline void init(IndexType* indices_, DistanceType* dists_) {
93  indices = indices_;
94  dists = dists_;
95  count = 0;
96  if (capacity)
97  dists[capacity - 1] = (std::numeric_limits<DistanceType>::max)();
98  }
99 
100  inline CountType size() const { return count; }
101 
102  inline bool full() const { return count == capacity; }
103 
104 
105  inline void addPoint(DistanceType dist, IndexType index) {
106  CountType i;
107  for (i = count; i > 0; --i) {
108 #ifdef NANOFLANN_FIRST_MATCH // If defined and two points have the same distance, the one with the
109  // lowest-index will be returned first.
110  if ((dists[i - 1] > dist) || ((dist == dists[i - 1]) && (indices[i - 1] > index))) {
111 #else
112  if (dists[i - 1] > dist) {
113 #endif
114  if (i < capacity) {
115  dists[i] = dists[i - 1];
116  indices[i] = indices[i - 1];
117  }
118  } else
119  break;
120  }
121  if (i < capacity) {
122  dists[i] = dist;
123  indices[i] = index;
124  }
125  if (count < capacity)
126  count++;
127  }
128 
129  inline DistanceType worstDist() const { return dists[capacity - 1]; }
130  };

253  inline DistanceType operator()(const T* a,
254  const size_t b_idx,
255  size_t size,
256  DistanceType worst_dist = -1) const {
257  DistanceType result = DistanceType();
258  const T* last = a + size;
259  const T* lastgroup = last - 3;
260  size_t d = 0;
261 
262  /* Process 4 items with each loop for efficiency. */
263  while (a < lastgroup) {
264  const DistanceType diff0 = std::abs(a[0] - data_source.kdtree_get_pt(b_idx, d++));
265  const DistanceType diff1 = std::abs(a[1] - data_source.kdtree_get_pt(b_idx, d++));
266  const DistanceType diff2 = std::abs(a[2] - data_source.kdtree_get_pt(b_idx, d++));
267  const DistanceType diff3 = std::abs(a[3] - data_source.kdtree_get_pt(b_idx, d++));
268  result += diff0 + diff1 + diff2 + diff3;
269  a += 4;
270  if ((worst_dist > 0) && (result > worst_dist)) {
271  return result;
272  }
273  }
274  /* Process last 0-3 components. Not needed for standard vector lengths. */
275  while (a < last) {
276  result += std::abs(*a++ - data_source.kdtree_get_pt(b_idx, d++));
277  }
278  return result;
279  }

449  class PooledAllocator {
450  /* We maintain memory alignment to word boundaries by requiring that all
451  allocations be in multiples of the machine wordsize. */
452  /* Size of machine word in bytes. Must be power of 2. */
453  /* Minimum number of bytes requested at a time from the system. Must be multiple of WORDSIZE. */
454 
455 
456  size_t remaining; /* Number of bytes left in current block of storage. */
457  void* base; /* Pointer to base of current block of storage. */
458  void* loc; /* Current location in block to next allocate memory. */
459 
460  void internal_init() {
461  remaining = 0;
462  base = NULL;
463  usedMemory = 0;
464  wastedMemory = 0;
465  }
466 
467  public:
468  size_t usedMemory;
469  size_t wastedMemory;
470 
474  PooledAllocator() { internal_init(); }
475 
479  ~PooledAllocator() { free_all(); }
480 
482  void free_all() {
483  while (base != NULL) {
484  void* prev = *(static_cast<void**>(base)); /* Get pointer to prev block. */
485  ::free(base);
486  base = prev;
487  }
488  internal_init();
489  }
490 
495  void* malloc(const size_t req_size) {
496  /* Round size up to a multiple of wordsize. The following expression
497  only works for WORDSIZE that is a power of 2, by masking last bits of
498  incremented size to zero.
499  */
500  const size_t size = (req_size + (WORDSIZE - 1)) & ~(WORDSIZE - 1);
501 
502  /* Check whether a new block must be allocated. Note that the first word
503  of a block is reserved for a pointer to the previous block.
504  */
505  if (size > remaining) {
506 
507  wastedMemory += remaining;
508 
509  /* Allocate new storage. */
510  const size_t blocksize = (size + sizeof(void*) + (WORDSIZE - 1) > BLOCKSIZE)
511  ? size + sizeof(void*) + (WORDSIZE - 1)
512  : BLOCKSIZE;
513 
514  // use the standard C malloc to allocate memory
515  void* m = ::malloc(blocksize);
516  if (!m) {
517  fprintf(stderr, "Failed to allocate memory.\n");
518  return NULL;
519  }
520 
521  /* Fill first word of new block with pointer to previous block. */
522  static_cast<void**>(m)[0] = base;
523  base = m;
524 
525  size_t shift = 0;
526  // int size_t = (WORDSIZE - ( (((size_t)m) + sizeof(void*)) & (WORDSIZE-1))) & (WORDSIZE-1);
527 
528  remaining = blocksize - sizeof(void*) - shift;
529  loc = (static_cast<char*>(m) + sizeof(void*) + shift);
530  }
531  void* rloc = loc;
532  loc = static_cast<char*>(loc) + size;
533  remaining -= size;
534 
535  usedMemory += size;
536 
537  return rloc;
538  }

1120  void middleSplit_(IndexType* ind,
1121  IndexType count,
1122  IndexType& index,
1123  int& cutfeat,
1124  DistanceType& cutval,
1125  const BoundingBox& bbox) {
1126  const DistanceType EPS = static_cast<DistanceType>(0.00001);
1127  ElementType max_span = bbox[0].high - bbox[0].low;
1128  for (int i = 1; i < (DIM > 0 ? DIM : dim); ++i) {
1129  ElementType span = bbox[i].high - bbox[i].low;
1130  if (span > max_span) {
1131  max_span = span;
1132  }
1133  }
1134  ElementType max_spread = -1;
1135  cutfeat = 0;
1136  for (int i = 0; i < (DIM > 0 ? DIM : dim); ++i) {
1137  ElementType span = bbox[i].high - bbox[i].low;
1138  if (span > (1 - EPS) * max_span) {
1139  ElementType min_elem, max_elem;
1140  computeMinMax(ind, count, cutfeat, min_elem, max_elem);
1141  ElementType spread = max_elem - min_elem;
1142  ;
1143  if (spread > max_spread) {
1144  cutfeat = i;
1145  max_spread = spread;
1146  }
1147  }
1148  }
1149  // split in the middle
1150  DistanceType split_val = (bbox[cutfeat].low + bbox[cutfeat].high) / 2;
1151  ElementType min_elem, max_elem;
1152  computeMinMax(ind, count, cutfeat, min_elem, max_elem);
1153 
1154  if (split_val < min_elem)
1155  cutval = min_elem;
1156  else if (split_val > max_elem)
1157  cutval = max_elem;
1158  else
1159  cutval = split_val;
1160 
1161  IndexType lim1, lim2;
1162  planeSplit(ind, count, cutfeat, cutval, lim1, lim2);
1163 
1164  if (lim1 > count / 2)
1165  index = lim1;
1166  else if (lim2 < count / 2)
1167  index = lim2;
1168  else
1169  index = count / 2;
1170  }

1182  void planeSplit(IndexType* ind,
1183  const IndexType count,
1184  int cutfeat,
1185  DistanceType& cutval,
1186  IndexType& lim1,
1187  IndexType& lim2) {
1188  /* Move vector indices for left subtree to front of list. */
1189  IndexType left = 0;
1190  IndexType right = count - 1;
1191  for (;;) {
1192  while (left <= right && dataset_get(ind[left], cutfeat) < cutval)
1193  ++left;
1194  while (right && left <= right && dataset_get(ind[right], cutfeat) >= cutval)
1195  --right;
1196  if (left > right || !right)
1197  break; // "!right" was added to support unsigned Index types
1198  std::swap(ind[left], ind[right]);
1199  ++left;
1200  --right;
1201  }
1202  /* If either list is empty, it means that all remaining features
1203  * are identical. Split in the middle to maintain a balanced tree.
1204  */
1205  lim1 = left;
1206  right = count - 1;
1207  for (;;) {
1208  while (left <= right && dataset_get(ind[left], cutfeat) <= cutval)
1209  ++left;
1210  while (right && left <= right && dataset_get(ind[right], cutfeat) > cutval)
1211  --right;
1212  if (left > right || !right)
1213  break; // "!right" was added to support unsigned Index types
1214  std::swap(ind[left], ind[right]);
1215  ++left;
1216  --right;
1217  }
1218  lim2 = left;
1219  }

1242  template <class RESULTSET>
1243  void searchLevel(RESULTSET& result_set,
1244  const Vector& vec,
1245  const NodePtr node,
1246  DistanceType mindistsq,
1247  distance_vector_t& dists,
1248  const float epsError) const {
1249  /* If this is a leaf node, then do check and return. */
1250  if ((node->child1 == NULL) && (node->child2 == NULL)) {
1251  // count_leaf += (node->lr.right-node->lr.left); // Removed since was neither used nor
1252  // returned to the user.
1253  DistanceType worst_dist = result_set.worstDist();
1254  for (IndexType i = node->node_type.lr.left; i < node->node_type.lr.right; ++i) {
1255  const IndexType index = vind[i]; // reorder... : i;
1256  DistanceType dist = distance(vec, index, (DIM > 0 ? DIM : dim));
1257  if (dist < worst_dist) {
1258  result_set.addPoint(dist, vind[i]);
1259  }
1260  }
1261  return;
1262  }
1263 
1264  /* Which child branch should be taken first? */
1265  int idx = node->node_type.sub.divfeat;
1266  ElementType val = vec[idx];
1267  DistanceType diff1 = val - node->node_type.sub.divlow;
1268  DistanceType diff2 = val - node->node_type.sub.divhigh;
1269 
1270  NodePtr bestChild;
1271  NodePtr otherChild;
1272  DistanceType cut_dist;
1273  if ((diff1 + diff2) < 0) {
1274  bestChild = node->child1;
1275  otherChild = node->child2;
1276  cut_dist = distance.accum_dist(val, node->node_type.sub.divhigh, idx);
1277  } else {
1278  bestChild = node->child2;
1279  otherChild = node->child1;
1280  cut_dist = distance.accum_dist(val, node->node_type.sub.divlow, idx);
1281  }
1282 
1283  /* Call recursively to search next level down. */
1284  searchLevel(result_set, vec, bestChild, mindistsq, dists, epsError);
1285 
1286  DistanceType dst = dists[idx];
1287  mindistsq = mindistsq + cut_dist - dst;
1288  dists[idx] = cut_dist;
1289  if (mindistsq * epsError <= result_set.worstDist()) {
1290  searchLevel(result_set, vec, otherChild, mindistsq, dists, epsError);
1291  }
1292  dists[idx] = dst;
1293  }
